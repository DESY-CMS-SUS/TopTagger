316  if(vecSubjetsMult_) subjets.back().setExtraVar("mult", (*vecSubjetsMult_) [iJet][iSJ]);
317  if(vecSubjetsPtD_) subjets.back().setExtraVar("ptD", (*vecSubjetsPtD_) [iJet][iSJ]);
318  if(vecSubjetsAxis1_) subjets.back().setExtraVar("axis1", (*vecSubjetsAxis1_)[iJet][iSJ]);
319  if(vecSubjetsAxis2_) subjets.back().setExtraVar("axis2", (*vecSubjetsAxis2_)[iJet][iSJ]);
320  }
321  }
322  else if (subjetsLVec_ != nullptr)
323  {
324  // Calculate matching subjets if a single list was given
325  for(unsigned int iSJ = 0; iSJ < subjetsLVec_->size(); ++iSJ)
326  {
327  double myDR = ROOT::Math::VectorUtil::DeltaR((*jetsLVec_)[iJet], (*subjetsLVec_)[iSJ]);
328  if (myDR < 0.8)
329  {
330  subjets.emplace_back((*subjetsLVec_)[iSJ], Constituent::AK8SUBJET);
331  if(subjetsBtag_) subjets.back().setBTag((*subjetsBtag_)[iSJ]);
332  if(subjetsMult_) subjets.back().setExtraVar("mult", (*subjetsMult_)[iSJ]);
333  if(subjetsPtD_) subjets.back().setExtraVar("ptD", (*subjetsPtD_)[iSJ]);
334  if(subjetsAxis1_) subjets.back().setExtraVar("axis1", (*subjetsAxis1_)[iSJ]);
335  if(subjetsAxis2_) subjets.back().setExtraVar("axis2", (*subjetsAxis2_)[iSJ]);
336  }
337  }
338 
339  // If more than 2 matches, find the best combination of two subjets
340  if (subjets.size() > 2)
341  {
342  double min_diff = 999999.;
343  int min_j=0, min_k=1;
344  for (unsigned int j=0 ; j<subjets.size(); ++j)
345  {
346  for (unsigned int k=j+1; k<subjets.size(); ++k)
347  {
348  TLorentzVector diff_LV = (*jetsLVec_)[iJet] - subjets[j].p() - subjets[k].p();
349  double diff = fabs(diff_LV.M());
350  if(diff < min_diff)
351  {
352  min_diff = diff;
353  min_j = j;
354  min_k = k;
355  }
356  }
357  }
358  subjets = {subjets[min_j], subjets[min_k]};
359  }
360  }
361  //Emplace new constituent into vector
362  if(tau1_ && tau2_ && tau3_)
363  {
364  constituents.emplace_back((*jetsLVec_)[iJet], static_cast<double>((*tau1_)[iJet]), static_cast<double>((*tau2_)[iJet]), static_cast<double>((*tau3_)[iJet]), static_cast<double>((*softDropMass_)[iJet]), subjets, getPUPPIweight(static_cast<double>((*jetsLVec_)[iJet].Pt()), static_cast<double>((*jetsLVec_)[iJet].Eta())));
365  }
366  else
367  {
368  constituents.emplace_back((*jetsLVec_)[iJet], 0.0, 0.0, 0.0, static_cast<double>((*softDropMass_)[iJet]), subjets, getPUPPIweight(static_cast<double>((*jetsLVec_)[iJet].Pt()), static_cast<double>((*jetsLVec_)[iJet].Eta())));
369  }
370 
371  if(deepAK8Top_)
372  {
373  constituents.back().setTopDisc((*deepAK8Top_)[iJet]);
374  }
375 
376  if(deepAK8W_)
377  {
378  constituents.back().setWDisc((*deepAK8W_)[iJet]);
379  }
380 
381  //Get gen matches if the required info is provided
382  if(hadGenTops_ && hadGenTopDaughters_)
383  {
384  for(unsigned int iGenTop = 0; iGenTop < hadGenTops_->size(); ++iGenTop)
385  {
386  for(const auto& genDaughter : (*hadGenTopDaughters_)[iGenTop])
387  {
388  for(const auto& subjet : subjets)
389  {
390  double dR = ROOT::Math::VectorUtil::DeltaR(subjet.p(), *genDaughter);
391  if(dR < 0.4)
392  {
393  constituents.back().addGenMatch((*hadGenTops_)[iGenTop], genDaughter);
394  }
395  }
396  }
397  }
398  }
399  }
400  }
